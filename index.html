function openPlayer(id) { const w = 1280; const h = 720; const left = (window.screen.width / 2) - (w / 2); const top = (window.screen.height / 2) - (h / 2); // HACK : mute=1 force l'autoplay immédiat sur Chrome, LibreWolf et Safari // autoplay=1 + playlist=${id} + loop=1 aide aussi à bypasser les restrictions const url = `https://www.youtube-nocookie.com/embed/${id}?autoplay=1&mute=1&rel=0&showinfo=0&modestbranding=1&enablejsapi=1&playlist=${id}`; const playerWin = window.open('', 'DCK_STUDIO_PLAYER', `toolbar=no, location=no, status=no, menubar=no, scrollbars=no, resizable=yes, width=${w}, height=${h}, top=${top}, left=${left}` ); // Si la fenêtre est déjà ouverte, on la met au premier plan if (playerWin) { playerWin.location.href = url; playerWin.focus(); } }
